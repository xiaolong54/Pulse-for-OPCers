```python
import httpx
import json
from sqlalchemy.orm import Session
from app.models.api_source import ApiSource
from app.models.sync_log import SyncLog
from app.models.metric import Metric
from app.core.database import SessionLocal
from datetime import datetime

async def sync_api_source(source_id: int) -> dict:
    db = SessionLocal()
    try:
        source = db.query(ApiSource).filter(ApiSource.id == source_id).first()
        if not source:
            return {"status": "failed", "error": "not found"}

        headers = source.headers or {}
        async with httpx.AsyncClient() as client:
            if source.method == "GET":
                resp = await client.get(source.endpoint, headers=headers)
            else:
                resp = await client.post(source.endpoint, headers=headers)
            data = resp.json()

        records = data if isinstance(data, list) else data.get("data", [])
        count = len(records)

        log = SyncLog(source_id=source_id, sync_type="manual", status="success", records_count=count)
        db.add(log)
        source.last_sync_at = datetime.now()
        db.commit()
        return {"status": "success", "records_count": count}
    except Exception as e:
        log = SyncLog(source_id=source_id, sync_type="manual", status="failed", error_message=str(e))
        db.add(log)
        db.commit()
        return {"status": "failed", "error": str(e)}
    finally:
        db.close()
```

```python
from fastapi import APIRouter, Depends
from pydantic import BaseModel
from sqlalchemy.orm import Session
from app.core.database import get_db
from app.models.api_source import ApiSource
from app.models.sync_log import SyncLog
from app.services.datasource_service import sync_api_source

router = APIRouter(prefix="/api/datasources", tags=["datasources"])

class SourceCreate(BaseModel):
    name: str
    endpoint: str
    method: str = "GET"
    auth_type: str = "none"
    auth_config: dict = {}
    headers: dict = {}
    data_mapping: dict = {}
    sync_cron: str = ""

@router.get("")
def list_sources(db: Session = Depends(get_db)):
    sources = db.query(ApiSource).order_by(ApiSource.created_at.desc()).all()
    return [{"id": s.id, "name": s.name, "endpoint": s.endpoint, "method": s.method,
             "auth_type": s.auth_type, "status": s.status, "last_sync_at": str(s.last_sync_at) if s.last_sync_at else "",
             "sync_cron": s.sync_cron, "created_at": str(s.created_at)[:10]} for s in sources]

@router.post("")
def create_source(req: SourceCreate, db: Session = Depends(get_db)):
    source = ApiSource(**req.dict())
    db.add(source)
    db.commit()
    return {"id": source.id, "name": source.name, "status": "created"}

@router.put("/{source_id}")
def update_source(source_id: int, req: SourceCreate, db: Session = Depends(get_db)):
    source = db.query(ApiSource).filter(ApiSource.id == source_id).first()
    if not source: return {"error": "not found"}
    for k, v in req.dict().items():
        setattr(source, k, v)
    db.commit()
    return {"ok": True}

@router.delete("/{source_id}")
def delete_source(source_id: int, db: Session = Depends(get_db)):
    db.query(ApiSource).filter(ApiSource.id == source_id).delete()
    db.commit()
    return {"ok": True}

@router.post("/{source_id}/sync")
async def sync_source(source_id: int, db: Session = Depends(get_db)):
    return await sync_api_source(source_id)

@router.get("/{source_id}/logs")
def get_logs(source_id: int, db: Session = Depends(get_db)):
    logs = db.query(SyncLog).filter(SyncLog.source_id == source_id).order_by(SyncLog.created_at.desc()).limit(20).all()
    return [{"id": l.id, "sync_type": l.sync_type, "status": l.status,
             "records_count": l.records_count, "error_message": l.error_message,
             "created_at": str(l.created_at)} for l in logs]
```

```python
from sqlalchemy import create_engine, text
from app.models.db_source import DbSource
from app.models.metric import Metric
from app.core.database import SessionLocal
from datetime import datetime

def test_connection(source: DbSource) -> dict:
    try:
        url = f"mysql+pymysql://{source.username}:{source.password_encrypted}@{source.host}:{source.port}/{source.database_name}"
        engine = create_engine(url)
        with engine.connect() as conn:
            conn.execute(text("SELECT 1"))
        return {"status": "success"}
    except Exception as e:
        return {"status": "failed", "error": str(e)}

def sync_from_db(source_id: int) -> dict:
    db = SessionLocal()
    try:
        source = db.query(DbSource).filter(DbSource.id == source_id).first()
        if not source:
            return {"status": "failed", "error": "not found"}

        url = f"mysql+pymysql://{source.username}:{source.password_encrypted}@{source.host}:{source.port}/{source.database_name}"
        engine = create_engine(url)

        with engine.connect() as conn:
            result = conn.execute(text(f"SELECT * FROM {source.table_name}"))
            rows = result.fetchall()
            columns = result.keys()

        count = len(rows)
        source.last_sync_at = datetime.now()
        db.commit()
        return {"status": "success", "records_count": count, "columns": list(columns)}
    except Exception as e:
        return {"status": "failed", "error": str(e)}
    finally:
        db.close()

def get_tables(source: DbSource) -> list:
    try:
        url = f"mysql+pymysql://{source.username}:{source.password_encrypted}@{source.host}:{source.port}/{source.database_name}"
        engine = create_engine(url)
        with engine.connect() as conn:
            result = conn.execute(text("SHOW TABLES"))
            return [row[0] for row in result]
    except:
        return []
```

```tsx
      title: '操作', key: 'actions',
      render: (_: any, r: any) => (
        <Space>
          <Button size="small" icon={<CheckCircleOutlined />} onClick={() => handleTest(r.id)}>测试</Button>
          <Button size="small" icon={<SyncOutlined />} onClick={() => handleSync(r.id)}>同步</Button>
          <Popconfirm title="确定删除？" onConfirm={() => handleDelete(r.id)}>
            <Button size="small" danger icon={<DeleteOutlined />} />
          </Popconfirm>
        </Space>
      )
    }
  ]

  return (
    <div style={{ padding: 24 }}>
      <Card title="数据库连接" extra={<Button type="primary" icon={<PlusOutlined />} onClick={() => setModalOpen(true)}>新建连接</Button>}>
        <Table rowKey="id" columns={columns} dataSource={sources} loading={loading} pagination={{ pageSize: 10 }} />
      </Card>

      <Modal title="新建数据库连接" open={modalOpen} onCancel={() => setModalOpen(false)} onOk={() => form.submit()} width={500}>
        <Form form={form} layout="vertical" onFinish={handleCreate}>
          <Form.Item name="name" label="名称" rules={[{ required: true }]}><Input /></Form.Item>
          <Form.Item name="host" label="主机地址" rules={[{ required: true }]} initialValue="localhost"><Input /></Form.Item>
          <Form.Item name="port" label="端口" initialValue={3306}><InputNumber style={{ width: '100%' }} /></Form.Item>
          <Form.Item name="database_name" label="数据库名" rules={[{ required: true }]}><Input /></Form.Item>
          <Form.Item name="username" label="用户名"><Input /></Form.Item>
          <Form.Item name="password" label="密码"><Input.Password /></Form.Item>
          <Form.Item name="table_name" label="表名"><Input placeholder="留空则读取所有表" /></Form.Item>
        </Form>
      </Modal>
    </div>
  )
}
```
