`backend/app/services/import_service.py`:

```python
import pandas as pd
import json
from io import BytesIO

def parse_file(file_content: bytes, filename: str) -> list[dict]:
    ext = filename.split('.')[-1].lower()
    if ext in ['xlsx', 'xls']:
        df = pd.read_excel(BytesIO(file_content))
    elif ext == 'csv':
        df = pd.read_csv(BytesIO(file_content))
    elif ext == 'json':
        return json.loads(file_content)
    else:
        raise ValueError(f"不支持的文件格式: {ext}")
    return df.to_dict('records')

def preview_data(records: list[dict]) -> dict:
    if not records:
        return {"columns": [], "rows": [], "total": 0}
    columns = list(records[0].keys())
    return {"columns": columns, "rows": records[:10], "total": len(records)}
```

- [ ] **Step 3: 创建导入API**

`backend/app/api/import.py`:
```python
from fastapi import APIRouter, UploadFile, File, Depends
from pydantic import BaseModel
from sqlalchemy.orm import Session
from app.core.database import get_db
from app.models.import_record import ImportRecord
from app.models.metric import Metric
from app.services.import_service import parse_file, preview_data
from datetime import datetime

router = APIRouter(prefix="/api/import", tags=["import"])

@router.post("/upload")
async def upload_file(file: UploadFile = File(...), db: Session = Depends(get_db)):
    content = await file.read()
    try:
        records = parse_file(content, file.filename)
        preview = preview_data(records)
        record = ImportRecord(
            filename=file.filename,
            file_type=file.filename.split('.')[-1],
            file_size=len(content),
            records_count=len(records),
            status="pending"
        )
        db.add(record)
        db.commit()
        return {
            "id": record.id,
            "filename": file.filename,
            "preview": preview,
            "total_records": len(records)
        }
    except Exception as e:
        return {"error": str(e)}

@router.post("/confirm/{import_id}")
def confirm_import(import_id: int, db: Session = Depends(get_db)):
    record = db.query(ImportRecord).filter(ImportRecord.id == import_id).first()
    if not record:
        return {"error": "not found"}
    record.status = "success"
    db.commit()
    return {"ok": True, "message": f"成功导入{record.records_count}条数据"}

@router.get("/records")
def list_imports(db: Session = Depends(get_db)):
    results = db.query(ImportRecord).order_by(ImportRecord.created_at.desc()).all()
    return [{"id": r.id, "filename": r.filename, "file_type": r.file_type,
             "records_count": r.records_count, "status": r.status,
             "created_at": str(r.created_at)[:19]} for r in results]
```

```ts
import api from './api'

export async function uploadFile(file: File) {
  const formData = new FormData()
  formData.append('file', file)
  const { data } = await api.post('/import/upload', formData, {
    headers: { 'Content-Type': 'multipart/form-data' }
  })
  return data
}

export async function confirmImport(id: number) {
  const { data } = await api.post(`/import/confirm/${id}`)
  return data
}

export async function getImportRecords() {
  const { data } = await api.get('/import/records')
  return data
}
```

- [ ] **Step 2: 创建文件导入页面**

`frontend/src/pages/Import.tsx`:
```tsx
import { useState, useEffect } from 'react'
import { Card, Upload, Button, Table, message, Typography, Space, Tag, Modal, Select } from 'antd'
import { InboxOutlined, CheckCircleOutlined, DeleteOutlined, EyeOutlined } from '@ant-design/icons'
import { uploadFile, confirmImport, getImportRecords } from '../services/import'

const { Dragger } = Upload

export default function Import() {
  const [records, setRecords] = useState([])
  const [loading, setLoading] = useState(false)
  const [preview, setPreview] = useState<any>(null)
  const [previewOpen, setPreviewOpen] = useState(false)

  useEffect(() => { loadRecords() }, [])

  const loadRecords = async () => {
    setLoading(true)
    const data = await getImportRecords()
    setRecords(data)
    setLoading(false)
  }

  const handleUpload = async (file: File) => {
    try {
      const result = await uploadFile(file)
      if (result.error) {
        message.error(result.error)
        return
      }
      message.success(`文件上传成功，共${result.total_records}条数据`)
      setPreview(result.preview)
      setPreviewOpen(true)
      loadRecords()
    } catch {
      message.error('上传失败')
    }
  }

  const handleConfirm = async (id: number) => {
    await confirmImport(id)
    message.success('导入已确认')
    loadRecords()
  }

  const columns = [
    { title: '文件名', dataIndex: 'filename', key: 'filename' },
    { title: '格式', dataIndex: 'file_type', key: 'file_type' },
    { title: '记录数', dataIndex: 'records_count', key: 'records_count' },
    { title: '状态', dataIndex: 'status', key: 'status', render: (s: string) => <Tag color={s === 'success' ? 'green' : s === 'failed' ? 'red' : 'orange'}>{s === 'success' ? '已完成' : s === 'failed' ? '失败' : '待确认'}</Tag> },
    { title: '时间', dataIndex: 'created_at', key: 'created_at' },
    {
      title: '操作', key: 'actions',
      render: (_: any, r: any) => r.status === 'pending' && <Button size="small" icon={<CheckCircleOutlined />} onClick={() => handleConfirm(r.id)}>确认导入</Button>
    }
  ]

  return (
    <div style={{ padding: 24 }}>
      <Card title="数据导入" style={{ marginBottom: 16 }}>
        <Dragger
          accept=".xlsx,.xls,.csv,.json"
          showUploadList={false}
          beforeUpload={(file) => { handleUpload(file); return false }}
        >
          <p className="ant-upload-drag-icon"><InboxOutlined /></p>
          <Typography.Text>点击或拖拽文件到此区域上传</Typography.Text>
          <Typography.Paragraph type="secondary">支持 Excel (.xlsx, .xls), CSV (.csv), JSON (.json) 格式</Typography.Paragraph>
        </Dragger>
      </Card>

      <Card title="导入记录">
        <Table rowKey="id" columns={columns} dataSource={records} loading={loading} pagination={{ pageSize: 10 }} />
      </Card>

      <Modal title="数据预览" open={previewOpen} onCancel={() => setPreviewOpen(false)} footer={null} width={800}>
        {preview && (
          <div>
            <Typography.Text>共 {preview.total} 条数据</Typography.Text>
            <Table
              rowKey={() => Math.random()}
              columns={preview.columns.map((c: string) => ({ title: c, dataIndex: c, key: c }))}
              dataSource={preview.rows.map((r: any, i: number) => ({ ...r, _key: i }))}
              pagination={false}
              scroll={{ x: 'max-content' }}
              style={{ marginTop: 16 }}
            />
          </div>
        )}
      </Modal>
    </div>
  )
}
```
