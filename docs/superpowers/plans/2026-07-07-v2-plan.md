```python
from fastapi import APIRouter, Depends
from pydantic import BaseModel
from sqlalchemy.orm import Session
from app.core.database import get_db
from app.models.report import Report
from app.services.report_service import generate_report_content
from app.models.metric import Metric
from sqlalchemy import func

router = APIRouter(prefix="/api/reports", tags=["reports"])

class GenerateRequest(BaseModel):
    type: str  # monthly/quarterly/annual/custom
    title: str = ""
    data_range: str = ""

@router.get("")
def list_reports(db: Session = Depends(get_db)):
    results = db.query(Report).order_by(Report.created_at.desc()).all()
    return [
        {"id": r.id, "title": r.title, "type": r.type, "format": r.format,
         "generated_by": r.generated_by, "created_at": str(r.created_at)[:10]}
        for r in results
    ]

@router.post("/generate")
async def generate(req: GenerateRequest, db: Session = Depends(get_db)):
    total = db.query(func.sum(Metric.value)).scalar() or 0
    count = db.query(Metric).count()
    data_context = f"数据总值：{total:.0f}，数据点数：{count}"
    content = await generate_report_content(req.type, data_context)
    title = req.title or f"{'月度' if req.type=='monthly' else '季度'}报告 {datetime.now().strftime('%Y-%m')}"
    report = Report(title=title, type=req.type, content=content, generated_by="ai", format="markdown")
    db.add(report)
    db.commit()
    return {"id": report.id, "title": title, "content": content}

@router.get("/{report_id}")
def get_report(report_id: int, db: Session = Depends(get_db)):
    r = db.query(Report).filter(Report.id == report_id).first()
    if not r:
        return {"error": "not found"}
    return {"id": r.id, "title": r.title, "content": r.content, "type": r.type, "format": r.format, "created_at": str(r.created_at)}

@router.delete("/{report_id}")
def delete_report(report_id: int, db: Session = Depends(get_db)):
    db.query(Report).filter(Report.id == report_id).delete()
    db.commit()
    return {"ok": True}
```

```python
from fastapi import APIRouter, Depends
from pydantic import BaseModel
from sqlalchemy.orm import Session
from app.core.database import get_db
from app.models.alert import Task
from datetime import date

router = APIRouter(prefix="/api/tasks", tags=["tasks"])

class TaskCreate(BaseModel):
    title: str
    description: str = ""
    project: str = ""
    assignee: str = ""
    priority: str = "medium"
    due_date: str = ""

@router.get("")
def list_tasks(project: str = "", status: str = "", db: Session = Depends(get_db)):
    q = db.query(Task)
    if project: q = q.filter(Task.project == project)
    if status: q = q.filter(Task.status == status)
    results = q.order_by(Task.created_at.desc()).all()
    return [{"id": r.id, "title": r.title, "project": r.project, "assignee": r.assignee,
             "priority": r.priority, "status": r.status, "due_date": str(r.due_date) if r.due_date else "",
             "estimated_hours": r.estimated_hours, "created_at": str(r.created_at)[:10]} for r in results]

@router.post("")
def create_task(req: TaskCreate, db: Session = Depends(get_db)):
    task = Task(title=req.title, description=req.description, project=req.project,
                assignee=req.assignee, priority=req.priority)
    if req.due_date: task.due_date = date.fromisoformat(req.due_date)
    db.add(task)
    db.commit()
    return {"id": task.id, "title": task.title, "status": "created"}

@router.put("/{task_id}")
def update_task(task_id: int, req: TaskCreate, db: Session = Depends(get_db)):
    task = db.query(Task).filter(Task.id == task_id).first()
    if not task: return {"error": "not found"}
    for k, v in req.dict().items():
        if v: setattr(task, k, v)
    if req.due_date: task.due_date = date.fromisoformat(req.due_date)
    db.commit()
    return {"ok": True}

@router.delete("/{task_id}")
def delete_task(task_id: int, db: Session = Depends(get_db)):
    db.query(Task).filter(Task.id == task_id).delete()
    db.commit()
    return {"ok": True}

@router.post("/{task_id}/complete")
def complete_task(task_id: int, db: Session = Depends(get_db)):
    task = db.query(Task).filter(Task.id == task_id).first()
    if task: task.status = "done"
    db.commit()
    return {"ok": True}

@router.get("/stats")
def task_stats(db: Session = Depends(get_db)):
    total = db.query(Task).count()
    todo = db.query(Task).filter(Task.status == "todo").count()
    in_progress = db.query(Task).filter(Task.status == "in_progress").count()
    done = db.query(Task).filter(Task.status == "done").count()
    return {"total": total, "todo": todo, "in_progress": in_progress, "done": done}
```
