`backend/app/services/ai_prediction.py`:
```python
from app.services.ai_service import chat_with_ai
from sqlalchemy.orm import Session
from app.models.metric import Metric
from sqlalchemy import func
from datetime import datetime, timedelta
import json

async def predict_trend(metric_name: str, days: int, db: Session) -> dict:
    since = datetime.now() - timedelta(days=days * 2)
    data = db.query(Metric).filter(
        Metric.name.like(f"%{metric_name}%"),
        Metric.recorded_at >= since
    ).order_by(Metric.recorded_at).all()

    data_str = "\n".join([f"{d.recorded_at.strftime('%Y-%m-%d')}: {d.value}" for d in data[-30:]])

    prompt = f"""基于以下历史数据，预测未来{days}天的{metric_name}趋势：
历史数据：
{data_str}

请以JSON格式返回，格式：{{"predictions": [{{"date": "2026-07-10", "value": 100, "lower": 90, "upper": 110}}]}}
需要包含未来{days}天每天的预测值、下限和上限。"""

    result = await chat_with_ai(prompt)
    # 从AI回复中提取JSON
    try:
        import re
        json_str = re.search(r'\{.*\}', result, re.DOTALL).group()
        return json.loads(json_str)
    except:
        return {"predictions": [{"date": str(datetime.now().date()), "value": 0, "lower": 0, "upper": 0}]}
```

- [ ] **Step 2: 创建趋势预测API**

`backend/app/api/ai_predict.py`:
```python
from fastapi import APIRouter, Depends, Query
from sqlalchemy.orm import Session
from app.core.database import get_db
from app.services.ai_prediction import predict_trend
from app.services.ai_service import chat_with_ai
from app.models.metric import Metric
from sqlalchemy import func
from datetime import datetime, timedelta

router = APIRouter(prefix="/api/ai", tags=["ai"])

@router.post("/predict")
async def predict(metric_name: str = Query("销售"), days: int = Query(30), db: Session = Depends(get_db)):
    return await predict_trend(metric_name, days, db)

@router.post("/analyze")
async def analyze(db: Session = Depends(get_db)):
    month_start = datetime.now().replace(day=1)
    total = db.query(func.sum(Metric.value)).filter(Metric.recorded_at >= month_start).scalar() or 0
    count = db.query(Metric).filter(Metric.recorded_at >= month_start).count()
    data_context = f"本月数据总值：{total:.0f}，数据点数：{count}"
    prompt = f"基于以下数据给出3-5条关键洞察和行动建议：{data_context}"
    result = await chat_with_ai(prompt)
    return {"analysis": result}

@router.post("/detect-anomaly")
async def detect_anomaly(db: Session = Depends(get_db)):
    since = datetime.now() - timedelta(days=30)
    data = db.query(Metric).filter(Metric.recorded_at >= since).all()
    data_str = "\n".join([f"{d.recorded_at.strftime('%Y-%m-%d')} {d.name}: {d.value}" for d in data[-50:]])
    prompt = f"分析以下数据，检测异常点，给出异常原因和处理建议：\n{data_str}"
    result = await chat_with_ai(prompt)
    return {"anomalies": result}

@router.post("/simulate")
async def simulate(decision: str = Query(...), db: Session = Depends(get_db)):
    month_start = datetime.now().replace(day=1)
    total = db.query(func.sum(Metric.value)).filter(Metric.recorded_at >= month_start).scalar() or 0
    data_context = f"当前数据总值：{total:.0f}"
    prompt = f"{data_context}\n如果执行以下决策：{decision}\n请模拟可能的结果并给出风险评估。"
    result = await chat_with_ai(prompt)
    return {"simulation": result}
```

```ts
import api from './api'

export async function predictTrend(metricName: string, days: number) {
  const { data } = await api.post('/ai/predict', null, { params: { metric_name: metricName, days } })
  return data
}

export async function analyzeData() {
  const { data } = await api.post('/ai/analyze')
  return data
}

export async function detectAnomaly() {
  const { data } = await api.post('/ai/detect-anomaly')
  return data
}

export async function simulateDecision(decision: string) {
  const { data } = await api.post(`/ai/simulate?decision=${encodeURIComponent(decision)}`)
  return data
}
```

- [ ] **Step 2: 创建AI分析页面**

`frontend/src/pages/AIAnalysis.tsx`:
```tsx
import { useState } from 'react'
import { Card, Row, Col, Select, InputNumber, Button, Typography, Space, Tag, Spin, message, Divider } from 'antd'
import { BulbOutlined, WarningOutlined, LineChartOutlined, ExperimentOutlined } from '@ant-design/icons'
import ReactEChartsCore from 'echarts-for-react'
import { predictTrend, analyzeData, detectAnomaly, simulateDecision } from '../services/ai_prediction'

const { TextArea } = Input

export default function AIAnalysis() {
  const [metricName, setMetricName] = useState('销售')
  const [days, setDays] = useState(30)
  const [predictLoading, setPredictLoading] = useState(false)
  const [predictData, setPredictData] = useState<any>(null)
  const [analyzeLoading, setAnalyzeLoading] = useState(false)
  const [analysis, setAnalysis] = useState('')
  const [anomalyLoading, setAnomalyLoading] = useState(false)
  const [anomalyResult, setAnomalyResult] = useState('')
  const [decision, setDecision] = useState('')
  const [simulateLoading, setSimulateLoading] = useState(false)
  const [simulation, setSimulation] = useState('')

  const handlePredict = async () => {
    setPredictLoading(true)
    try {
      const data = await predictTrend(metricName, days)
      setPredictData(data)
    } catch { message.error('预测失败') }
    setPredictLoading(false)
  }

  const handleAnalyze = async () => {
    setAnalyzeLoading(true)
    try {
      const data = await analyzeData()
      setAnalysis(data.analysis)
    } catch { message.error('分析失败') }
    setAnalyzeLoading(false)
  }

  const handleDetect = async () => {
    setAnomalyLoading(true)
    try {
      const data = await detectAnomaly()
      setAnomalyResult(data.anomalies)
    } catch { message.error('异常检测失败') }
    setAnomalyLoading(false)
  }

  const handleSimulate = async () => {
    if (!decision) return
    setSimulateLoading(true)
    try {
      const data = await simulateDecision(decision)
      setSimulation(data.simulation)
    } catch { message.error('模拟失败') }
    setSimulateLoading(false)
  }

  const chartOption = predictData?.predictions ? {
    tooltip: { trigger: 'axis' },
    xAxis: { type: 'category', data: predictData.predictions.map((d: any) => d.date) },
    yAxis: { type: 'value' },
    grid: { left: 60, right: 20, top: 20, bottom: 30 },
    series: [
      {
        type: 'line', name: '预测值',
        data: predictData.predictions.map((d: any) => d.value),
        smooth: true, lineStyle: { width: 2 }
      },
      {
        type: 'line', name: '上限',
        data: predictData.predictions.map((d: any) => d.upper),
        lineStyle: { opacity: 0.3 }, smooth: true
      },
      {
        type: 'line', name: '下限',
        data: predictData.predictions.map((d: any) => d.lower),
        lineStyle: { opacity: 0.3 }, smooth: true,
        areaStyle: { color: 'rgba(22,119,255,0.1)' }
      }
    ]
  } : null

  return (
    <div style={{ padding: 24 }}>
      <Row gutter={[16, 16]}>
        <Col span={24}>
          <Card title={<><LineChartOutlined /> 趋势预测</>}>
            <Space style={{ marginBottom: 16 }}>
              <Select value={metricName} onChange={setMetricName} style={{ width: 120 }}
                options={[{ label: '销售', value: '销售' }, { label: '成本', value: '成本' }, { label: '利润', value: '利润' }]} />
              <InputNumber value={days} onChange={v => setDays(v || 30)} addonAfter="天" min={1} max={365} />
              <Button type="primary" onClick={handlePredict} loading={predictLoading}>开始预测</Button>
            </Space>
            {chartOption && <ReactEChartsCore option={chartOption} style={{ height: 300 }} />}
          </Card>
        </Col>

        <Col span={12}>
          <Card title={<><BulbOutlined /> 智能分析建议</>}>
            <Button onClick={handleAnalyze} loading={analyzeLoading} style={{ marginBottom: 16 }}>生成分析</Button>
            {analysis && <Typography.Paragraph style={{ whiteSpace: 'pre-wrap' }}>{analysis}</Typography.Paragraph>}
          </Card>
        </Col>

        <Col span={12}>
          <Card title={<><WarningOutlined /> 异常检测</>}>
            <Button onClick={handleDetect} loading={anomalyLoading} style={{ marginBottom: 16 }}>检测异常</Button>
            {anomalyResult && <Typography.Paragraph style={{ whiteSpace: 'pre-wrap' }}>{anomalyResult}</Typography.Paragraph>}
          </Card>
        </Col>

        <Col span={24}>
          <Card title={<><ExperimentOutlined /> 决策模拟</>}>
            <Space direction="vertical" style={{ width: '100%' }}>
              <TextArea rows={3} value={decision} onChange={e => setDecision(e.target.value)}
                placeholder="输入你的决策方案，例如：加大华东区域市场投入20%" />
              <Button type="primary" onClick={handleSimulate} loading={simulateLoading}>模拟分析</Button>
            </Space>
            {simulation && <Typography.Paragraph style={{ whiteSpace: 'pre-wrap', marginTop: 16 }}>{simulation}</Typography.Paragraph>}
          </Card>
        </Col>
      </Row>
    </div>
  )
}
```
